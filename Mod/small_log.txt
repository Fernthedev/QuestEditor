void logBytes(std::string arg, void* ptr, int byteNum) {
    std::stringstream ss;
    char* bytes = reinterpret_cast<char*>(ptr);
    for(int i = 0; i < byteNum; i++) {
        char byte = bytes[i];
        for(int j = 0; j < 8; j++) {
            // darn negatives
            ss << (byte % 2)*(byte % 2);
            // remove last bit
            byte = byte >> 1;
        }
    }
    std::string str = ss.str();
    std::string rev(str.rbegin(), str.rend());
    LOG_INFO("%s = %s", arg.c_str(), rev.c_str());
}
